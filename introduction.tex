\begin{frame}{Hoare logic}
\centering
\LARGE
\begin{mathpar}
    \iSpec{
      \textcolor{color2}{\iProp}
    }{
      \textcolor{color1}{\zooExpr}
    }[
      \textcolor{color3}{\zooVal}
    ]{
      \textcolor{color2}{\iPropTwo}
    }
  \and
    \begin{array}{l}
        \text{precondition}
      \\
        \text{expression / instruction}
      \\
        \text{postcondition}
    \end{array}
\end{mathpar}
\vfill
\Large
If the initial state satisfies \textcolor{color2}{\iProp}, \\
\textcolor{color1}{\zooExpr} executes safely and returns a value \textcolor{color3}{\zooVal} \\
such that the final state satisfies \textcolor{color2}{\iPropTwo}.
\end{frame}

% refinement / extension of Hoare logic
% resources
% local reasoning on programs with pointers
% local reasoning on concurrent programs
% ownership transfers
% assertions are stable in the face of interference
% particularly suited for concurrency
\begin{frame}{Separation logic}
\centering
\Large
\begin{tabular}{lll}
    assuming facts & $\rightarrow$ & owning \textcolor{color2}{resources}
  \\
    global reasoning & $\rightarrow$ & local reasoning
\end{tabular}
\vfill
\LARGE
\begin{mathpar}
    \iSpec{
      \textcolor{color2}{\iProp}
    }{
      \zooExpr
    }[
      \zooVal
    ]{
      \textcolor{color2}{\iPropTwo}
    }
  \and
    \iSpec{
      \textcolor{color2}{\zooLoc \iPointsto \zooVal}
    }{
      !\zooLoc
    }[
      \v{res}
    ]{
      \v{res} = \zooVal \iSep
      \textcolor{color2}{\zooLoc \iPointsto \zooVal}
    }
\end{mathpar}
\vfill
\Large
Given \textcolor{color2}{resources \iProp}, \\
\zooExpr executes safely and returns a value \zooVal \\
and \textcolor{color2}{resources \iPropTwo}.
\end{frame}
