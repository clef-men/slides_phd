\section{\OCaml}

\begin{frame}{\includegraphics[scale=.15]{images/ocaml.png}}
\Large
\begin{itemize}
  \item Typically \emph{french} programming language (it's quite good).
\end{itemize}
\vfill
\begin{itemize}
  \item \textbf{Functional}: first-class functions, algebraic data types.
  \item \textbf{Imperative}: references, mutable records.
  \item \textbf{Strongly typed}: well-typedness $\implies$ safety.
  \item \textbf{Garbage-collected}: automatic memory management.
\end{itemize}
\vfill
\begin{itemize}
  \item \textbf{Parallel}: domains, atomic references, blocking mechanisms.
  \item \textbf{Concurrent}: algebraic effects.
\end{itemize}
\end{frame}
