  <<< 
    stack_inv t ι
  | ∀∀ vs,
    stack_model t vs 
  >>>
    stack_push t v @ ↑ι
  <<<
    stack_model t (v :: vs)
  | RET ();
    True
  >>>.
Proof.
  ...
Qed.
\end{minted}
